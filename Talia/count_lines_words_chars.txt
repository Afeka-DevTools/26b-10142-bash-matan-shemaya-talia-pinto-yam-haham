# פקודה: ספירת שורות, מילים, ותווים בכל קובץ בתיקיה והדפסתם
#!/bin/bash

TARGET_DIR="/path/to/directory"

echo "סופר שורות, מילים ותווים עבור הקבצים בתיקייה: $TARGET_DIR"
echo "--------------------------------------------------------"

for file in "$TARGET_DIR"/*; do
    if [ -f "$file" ]; then
        wc "$file" | awk '{print "קובץ: " $4 "\nשורות: " $1 " | מילים: " $2 " | תווים: " $3 "\n"}'
    fi
done
